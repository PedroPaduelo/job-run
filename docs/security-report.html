const allowedOrigins = (process.env.ALLOWED_ORIGINS || 'http://localhost:5173,http://localhost:3000').split(',');

// CORS com callback de validação
origin: (origin, callback) => {
  if (!origin) return callback(null, true); // Same-origin
  if (allowedOrigins.includes(origin) || process.env.NODE_ENV === 'development') {
    return callback(null, true);
  }
  callback(new Error('Not allowed by CORS'), false);
}

Verificação JWT

// auth.ts:130-152
export async function verifyJWT(token: string): Promise {
  const jwks = await getJWKS();
  const JWKS = jose.createLocalJWKSet(jwks);

  const validIssuers = [
    baseUrl, baseUrl + '/',
    baseUrl.replace('https://', 'http://'),
    baseUrl.replace('http://', 'https://'),
    'https://auth.service.api.nommand.com',
    'http://auth.service.api.nommand.com',
    'https://auth.service.nommand.com',
    'http://localhost:3000',
  ];

  const { payload } = await jose.jwtVerify(token, JWKS, {
    issuer: validIssuers,
  });
  return payload as JWTPayload;
}

6. Autorização

PropriedadeValor
MiddlewarerequirePermission()

Função hasPermission

// auth.ts:385-398
export function hasPermission(
  userPermissions: string[],
  requiredPermission: string
): boolean {
  // Admin tem tudo
  if (userPermissions.includes('*')) return true;

  // Wildcard (ex: jobs:* inclui jobs:create)
  const [resource, action] = requiredPermission.split(':');
  if (userPermissions.includes(`${resource}:*`)) return true;

  // Permissão exata
  return userPermissions.includes(requiredPermission);
}

Uso no Middleware

// server.ts:135-143
export function requirePermission(permission: string) {
  return async (request: FastifyRequest, reply: FastifyReply) => {
    const userPermissions = request.permissions || [];
    if (!hasPermission(userPermissions, permission)) {
      await logAudit({ /* ... */ status: 'DENIED' });
      return reply.status(403).send({ error: 'Forbidden', message: `Missing permission: ${permission}` });
    }
  };
}

9. Riscos Identificados

IP Spoofing MÉDIO

request.ip pode ser spoofado via header X-Forwarded-For ou X-Real-IP.

Impacto: Bypass de IP whitelist em API Keys.

API Key em Plaintext ALTO

API Keys são armazenadas sem hash. Vazamento do banco expõe todas as chaves.

Impacto: Comprometimento total se o banco for vazado.

allowedDomains não validado MÉDIO

Campo existe no modelo mas não é usado para validar callbacks.

Impacto: Callbacks podem ir para domínios não autorizados.

Rate limit por API Key MÉDIO

Campo rateLimit existe no modelo mas não é aplicado.

Impacto: Uma API Key pode consumir todo o rate limit da organização.

Rate limit fallback para IP MÉDIO

Se organizationId não existir, usa IP como key, permitindo bypass.

Impacto: Ataque de rate limit por spoofing de IP.

Múltiplos Issuers JWT BAIXO

Múltiplos issuers aceitos para flexibilidade, mas é uma superfície de ataque pequena.

Recomendação: Restringir em produção.

12. Recomendações

1. Hash de API Keys

Armazenar API Keys com hash (bcrypt ou Argon2) ao invés de plaintext.

// Usar: await bcrypt.hash(key, 10)
// Na verificação: await bcrypt.compare(inputKey, storedHash)
// E armazenar apenas o hash!

2. Rate Limit por API Key

Implementar rate limiting individual para cada API Key, não apenas global.

// Verificar apiKey.rateLimit antes de processar
if (apiKey.rateLimit) {
  const key = `ratelimit:apikey:${apiKey.id}`;
  const current = await redis.incr(key);
  if (current === 1) await redis.expire(key, 60);
  if (current > apiKey.rateLimit) return 429;
}

3. Validar allowedDomains em Callbacks

Usar o campo allowedDomains para validar URLs de callback antes de executar webhooks.

4. Configurar Trusted Proxy

Configurar Fastify para confiar apenas em proxies confiáveis para evitar IP spoofing.

import fastify from 'fastify';
const app = fastify({
  trustProxy: '127.0.0.1', // ou array de IPs confiáveis
});

5. Rotação de API Keys

Implementar rotação automática de API Keys com expiração e alerta de renovação.

6. Audit Logs Estruturados

Adicionar mais contexto aos logs: геолокация, device info, duração da requisição.

7. HTTPS

Configurar HTTPS via EasyPanel ou nginx. Verificar que NODE_ENV=production.

8. Limitar Issuers JWT

Em produção, restringir os issuers JWT aceitos apenas ao Auth Service real.

Error Handling & Sentry

Estratégias robustas de captura, monitoramento e recuperação de erros no JobSync

Visão Geral

O JobSync implementa uma estratégia de múltiplas camadas para tratamento de erros. Desde erros de validação da API até falhas catastróficas no banco de dados, cada tipo de erro é tratado apropriadamente, logado no Sentry e, quando possível, no AuditLog.

Error Handler Global (Fastify)

O Fastify usa um setErrorHandler global que captura todas as exceções não tratadas. O handler logs o erro no Sentry com contexto rico e retorna mensagens genéricas em produção para evitar vazamento de informações.

Implementação

// src/server.ts ou src/lib/error-handler.ts
import { FastifyInstance } from 'fastify';
import { sentryErrorHandler } from './sentry';
import { captureAuditLog } from './audit';

export function setErrorHandler(server: FastifyInstance) {
  server.setErrorHandler(async (error, request, reply) => {
    // Captura contexto da requisição
    const context = {
      url: request.url,
      method: request.method,
      organizationId: request.organizationId,
      userId: request.userId || request.apiKeyId,
      headers: {
        'user-agent': request.headers['user-agent'],
        'x-forwarded-for': request.headers['x-forwarded-for'],
      },
    };

    // Log no Sentry com contexto
    const sentryError = sentryErrorHandler(error, context);

    // Se há organizationId, log no AuditLog
    if (request.organizationId) {
      await captureAuditLog({
        organizationId: request.organizationId,
        action: 'error.occurred',
        userId: request.userId,
        metadata: {
          url: request.url,
          method: request.method,
          message: error.message,
          stack: process.env.NODE_ENV === 'development' ? error.stack : undefined,
          sentryId: sentryError?.eventId,
        },
        status: 'ERROR',
      });
    }

    // Define status code apropriado
    let statusCode = 500;
    if (error.statusCode) {
      statusCode = error.statusCode;
    } else if (error.name === 'ZodError') {
      statusCode = 400;
    } else if (error.name === 'UnauthorizedError') {
      statusCode = 401;
    } else if (error.name === 'ForbiddenError') {
      statusCode = 403;
    } else if (error.name === 'NotFoundError') {
      statusCode = 404;
    }

    // Mensagens diferenciadas por ambiente
    let message = 'Internal server error';
    if (process.env.NODE_ENV === 'development') {
      message = error.message;
    }

    reply.status(statusCode).send({
      error: {
        code: error.code || 'INTERNAL_ERROR',
        message,
        ...(process.env.NODE_ENV === 'development' && { stack: error.stack }),
      },
    });
  });
}

Configuração do Sentry

O Sentry é configurado para monitoramento de erros em tempo real com amostragem diferencial entre desenvolvimento e produção.

Inicialização

// src/lib/sentry.ts
import * as Sentry from '@sentry/node';
import { Integrations } from '@sentry/tracing';

export function initSentry() {
  const dsn = process.env.SENTRY_DSN;
  if (!dsn) {
    console.warn('SENTRY_DSN not set, Sentry disabled');
    return;
  }

  const isProduction = process.env.NODE_ENV === 'production';

  Sentry.init({
    dsn,
    environment: isProduction ? 'production' : 'development',
    tracesSampleRate: isProduction ? 0.1 : 1.0,
    beforeSend(event, hint) {
      // Em desenvolvimento, não envia eventos
      if (!isProduction) {
        console.log('[Sentry Event]', event, hint);
        return null;
      }
      return event;
    },
    beforeSendTransaction(context) {
      // Filtra transações de health check
      if (context.transaction?.name?.includes('/health')) {
        return null;
      }
      return context.transaction;
    },
    integrations: [
      new Integrations.Http({ tracing: true }),
      new Integrations.Express(),
      new Integrations.Console(),
    ],
  });
}

/**
 * Captura um erro no Sentry com contexto
 */
export function captureError(error: Error, context?: Record) {
  if (context) {
    Sentry.setContext('request', context);
  }
  return Sentry.captureException(error);
}

/**
 * Captura uma mensagem no Sentry
 */
export function captureMessage(message: string, level: 'info' | 'warning' | 'error' = 'info') {
  return Sentry.captureMessage(message, level);
}

/**
 * Associa um usuário ao contexto do Sentry
 */
export function setUser(user: { id: string; email?: string; organizationId?: string }) {
  Sentry.setUser({
    id: user.id,
    email: user.email,
    tags: user.organizationId ? { organizationId: user.organizationId } : undefined,
  });
}

/**
 * Adiciona um breadcrumb para rastrear a sequência de eventos
 */
export function addBreadcrumb(
  category: string,
  message: string,
  data?: Record
) {
  Sentry.addBreadcrumb({
    category,
    message,
    level: 'info',
    data,
    timestamp: new Date(),
  });
}

/**
 * Handler de erro do Fastify para integração com Sentry
 */
export async function sentryErrorHandler(
  error: Error,
  request?: { url?: string; method?: string }
) {
  const scope = Sentry.getCurrentHub().getScope();

  if (request) {
    scope.setTag('url', request.url || '');
    scope.setTag('method', request.method || '');
  }

  const eventId = Sentry.captureException(error);
  scope.clear();

  return { eventId };
}

/**
 * Flush do Sentry - usado no graceful shutdown
 */
export async function flushSentry(timeout = 2000): Promise<boolean> {
  return Sentry.close(timeout);
}

/**
 * Middleware para capturar erros de request
 */
export function sentryRequestHandler() {
  return Sentry.Handlers.requestHandler();
}

/**
 * Middleware para capturar erros de tracing
 */
export function sentryTracingHandler() {
  return Sentry.Handlers.tracingHandler();
}

Integração com Error Boundary

// frontend/src/components/ErrorBoundary.tsx
import React from 'react';
import * as Sentry from '@sentry/react';

interface Props {
  children: React.ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends React.Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    Sentry.withScope((scope) => {
      scope.setTag('component', 'ErrorBoundary');
      scope.setExtra('errorInfo', errorInfo);
      Sentry.captureException(error);
    });
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="error-boundary">
          <h2>Algo deu errado</h2>
          <p>{this.state.error?.message}</p>
          <button onClick={() => this.setState({ hasError: false })}>
            Tentar novamente
          </button>
        </div>
      );
    }

    return this.props.children;
  }
}

</Sentry.ErrorBoundary>

Graceful Shutdown

O sistema implementa desligamento gracefully para garantir que eventos em andamento no Sentry sejam enviados e conexões fechadas corretamente.

Implementação do Shutdown

// src/server.ts
import { flushSentry } from './lib/sentry';
import { prisma } from './lib/prisma';

// Signal handlers
process.on('SIGTERM', handleShutdown);
process.on('SIGINT', handleShutdown);

async function handleShutdown(signal: string) {
  console.log(`[Shutdown] Recebido ${signal}, iniciando graceful shutdown...`);

  try {
    // 1. Flush do Sentry (aguarda até 2 segundos)
    console.log('[Shutdown] Flushing Sentry...');
    await flushSentry(2000);
    console.log('[Shutdown] Sentry flushed');

    // 2. Fecha o Fastify
    console.log('[Shutdown] Closing Fastify server...');
    await server.close();
    console.log('[Shutdown] Fastify closed');

    // 3. Disconecta do Prisma
    console.log('[Shutdown] Disconnecting Prisma...');
    await prisma.$disconnect();
    console.log('[Shutdown] Prisma disconnected');

    // 4. Fecha workers BullMQ
    console.log('[Shutdown] Closing BullMQ workers...');
    await Promise.all(workers.map((w) => w.close()));
    console.log('[Shutdown] Workers closed');

    console.log('[Shutdown] Complete');
    process.exit(0);
  } catch (error) {
    console.error('[Shutdown] Error during shutdown:', error);
    process.exit(1);
  }
}

Shutdown de Workers

// src/workers/manager.ts
import { Worker } from 'bullmq';

const workers: Worker[] = [];

export function createWorker(queueName: string, concurrency: number) {
  const worker = new Worker(
    queueName,
    async (job) => {
      // Processa job
      return await processJob(job);
    },
    {
      connection: redisConfig,
      concurrency,
      autorun: false, // Inicia manualmente
    }
  );

  // Event listeners
  worker.on('completed', (job) => {
    console.log(`[Worker] Job ${job.id} completed`);
  });

  worker.on('failed', async (job, error) => {
    console.error(`[Worker] Job ${job.id} failed:`, error);

    // Log do erro no AuditLog
    await captureAuditLog({
      organizationId: job.data.organizationId,
      action: 'job.error',
      resourceId: job.id,
      resourceType: 'job',
      metadata: {
        queueName,
        failedReason: error.message,
        stack: error.stack,
        attempts: job.attemptsMade,
      },
      status: 'ERROR',
    });
  });

  worker.on('error', (error) => {
    console.error('[Worker] Fatal error:', error);

    // Worker travou - log crítico
    captureError(error, { queueName });
  });

  // Inicia o worker
  await worker.run();

  workers.push(worker);
  return worker;
}

Erros no Processamento de Jobs

Jobs falham por diversas razões: timeout, exceções no código, problemas de rede. O BullMQ gerencia retry automático e status de jobs.

Status de Jobs

StatusDescriçãoTransição
PENDING Aguardando na queue → WAITING, COMPLETED, FAILED
WAITING Worker disponível processando → ACTIVE, COMPLETED, FAILED
ACTIVE Em processamento → COMPLETED, FAILED
COMPLETED Processado com sucesso Estado final
FAILED Falhou (ainda com retries) → WAITING (retry) ou DEAD
DEAD Esgotou todos os retries Estado final
DELAYED Adiado para processamento futuro → WAITING quando delay expirar

Worker com JobLog

// src/workers/job-worker.ts
import { Job } from 'bullmq';

export async function processJob(job: Job<JobData>): Promise<JobResult> {
  const startTime = Date.now();
  const logParams = {
    jobId: job.id,
    organizationId: job.data.organizationId,
    queueName: job.queue.name,
    jobName: job.name,
  };

  // Registra início
  await prisma.jobLog.create({
    data: {
      ...logParams,
      status: 'ACTIVE',
      payload: job.data,
      startedAt: new Date(),
    },
  });

  try {
    // Processa o job
    const result = await executeJobLogic(job.data);

    const duration = Date.now() - startTime;

    // Registra sucesso
    await prisma.jobLog.update({
      where: { jobId: job.id },
      data: {
        status: 'COMPLETED',
        completedAt: new Date(),
        duration,
        result,
      },
    });

    addBreadcrumb('job', `Job ${job.id} completed`, {
      queueName: job.queue.name,
      duration,
    });

    return result;
  } catch (error: unknown) {
  const duration = Date.now() - startTime;
    const errorMessage = error instanceof Error ? error.message : String(error);

    // Registra falha
    await prisma.jobLog.update({
      where: { jobId: job.id },
      data: {
        status: 'FAILED',
        completedAt: new Date(),
        duration,
        Error: errorMessage,
        stackTrace: error instanceof Error ? error.stack : undefined,
        attemptsMade: job.attemptsMade,
      },
    });

    // Log no AuditLog
    await captureAuditLog({
      organizationId: job.data.organizationId,
      action: 'job.error',
      resourceId: job.id,
      resourceType: 'job',
      metadata: {
        queueName: job.queue.name,
        error: errorMessage,
        attempts: job.attemptsMade,
        stack: error instanceof Error ? error.stack : undefined,
      },
      status: 'ERROR',
    });

    // Re-lança para BullMQ gerenciar retry
    throw error;
  }
}

Unhandled Promise Rejections

// Capturar rejeições não tratadas globalmente
process.on('unhandledRejection', (reason, promise) => {
  console.error('[UnhandledRejection]', reason);

  const error = reason instanceof Error ? reason : new Error(String(reason));

  captureError(error, {
    type: 'unhandledRejection',
    promise: promise.toString(),
    timestamp: new Date().toISOString(),
  });

  // Em produção, tentar graceful shutdown após múltiplas rejeições
  if (process.env.NODE_ENV === 'production') {
    unhandledRejectionCount++;
    if (unhandledRejectionCount >= 3) {
      console.error('Múltiplas rejeições não tratadas, shutdown iniciado...');
      handleShutdown('unhandledRejection');
    }
  }
});

      );

      try {
        return await Promise.race([
          queries.execute(),
          timeoutPromise,
        ]);
      } catch (error) {
        captureError(error as Error, { query: queries.description });
        throw error;
      }
    },
  },
});

Memory Leaks

// Monitorar memória periodicamente
import { performance } from 'perf_hooks';

let lastMemoryUsage = process.memoryUsage();
let gcCount = 0;

// Forçar GC se disponível
const enableGCPressure = global.gc !== undefined;

setInterval(() => {
  const memory = process.memoryUsage();

  const heapDiff = memory.heapUsed - lastMemoryUsage.heapUsed;
  const heapGrowth = (heapDiff / lastMemoryUsage.heapUsed) * 100;

  if (heapGrowth > 10) {
    console.warn(`[Memory] Heap growth: ${heapGrowth.toFixed(2)}%`);

    captureMessage(`High memory growth: ${heapGrowth.toFixed(2)}%`, 'warning', {
      heapUsed: memory.heapUsed,
      heapTotal: memory.heapTotal,
      rss: memory.rss,
    });

    // Force garbage collection em dev
    if (enableGCPressure) {
      global.gc();
      gcCount++;
      console.log(`[Memory] GC forced (count: ${gcCount})`);
    }
  }

  lastMemoryUsage = memory;

  // Alertar se heapVeryApproachingLimit
  const heapLimit = v8.getHeapStatistics().heap_size_limit;
  if (memory.heapUsed > heapLimit * 0.9) {
    captureError(new Error('Memory usage approaching limit'), {
      heapUsed: memory.heapUsed,
      heapLimit,
      usagePercent: (memory.heapUsed / heapLimit) * 100,
    });
  }
}, 30000); // A cada 30 segundos

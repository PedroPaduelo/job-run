    }
  });

  // Sincronizar Membership

          userId: user.id,
          organizationId: org.id
        }
      },

        role: membership.role,
        permissions: membership.permissions,
        joinedAt: membership.joinedAt
      },

        userId: user.id,
        organizationId: org.id,
        role: membership.role,
        permissions: membership.permissions,
        joinedAt: membership.joinedAt
      }
    });
  }
}
